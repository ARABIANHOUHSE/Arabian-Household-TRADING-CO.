    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const countdownDate = new Date("Sep 16, 2025 00:00:00").getTime();

            const countdownFunction = setInterval(function() {
                const now = new Date().getTime();
                const distance = countdownDate - now;

                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);

                document.getElementById("days").innerText = days < 10 ? '0' + days : days;
                document.getElementById("hours").innerText = hours < 10 ? '0' + hours : hours;
                document.getElementById("minutes").innerText = minutes < 10 ? '0' + minutes : minutes;
                document.getElementById("seconds").innerText = seconds < 10 ? '0' + seconds : seconds;

                if (distance < 0) {
                    clearInterval(countdownFunction);
                    document.getElementById("countdown").innerHTML = "<div class='text-2xl font-bold highlight-text'>العرض بدأ الآن!</div>";
                }
            }, 1000);

            const tabs = document.querySelectorAll('.tab-button');
            const tabPanes = document.querySelectorAll('.tab-pane');

            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    tabs.forEach(item => item.classList.remove('active', 'text-gray-900', 'font-bold'));
                    tabs.forEach(item => item.classList.add('text-gray-500'));
                    
                    tab.classList.add('active', 'text-gray-900', 'font-bold');
                    tab.classList.remove('text-gray-500');

                    tabPanes.forEach(pane => {
                        pane.classList.add('hidden');
                    });
                    
                    const targetPane = document.getElementById(tab.dataset.tab);
                    if(targetPane) {
                        targetPane.classList.remove('hidden');
                    }
                });
            });

            const apiKey = "";
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;
            const ideaButton = document.getElementById('generate-idea');
            const ideaOutput = document.getElementById('idea-output');

            ideaButton.addEventListener('click', async () => {
                ideaButton.disabled = true;
                ideaButton.innerText = 'جاري التفكير...';
                ideaOutput.innerHTML = '<span class="text-gray-400">جاري تحميل فكرتك...</span>';

                const prompts = [
                    "اكتب فكرة لوصفة حلويات مبتكرة باستخدام أدوات مطبخ عصرية، واذكر اسم الوصفة.",
                    "اقتراح فكرة عشاء سريع ومميز يمكن تحضيره في أواني الطهي المتوفرة لدينا.",
                    "اكتب نصيحة قصيرة لتنظيم المطبخ باستخدام أواني التخزين الأنيقة.",
                    "اكتب فكرة لتقديم مشروبات ساخنة في المناسبات الشتوية باستخدام أكواب أنيقة.",
                    "اقتراح فكرة لتقديم طبق رئيسي بطريقة فنية ومختلفة باستخدام أواني السيراميك الفاخرة."
                ];

                const randomPrompt = prompts[Math.floor(Math.random() * prompts.length)];

                const payload = {
                    contents: [{ parts: [{ text: randomPrompt }] }],
                    systemInstruction: {
                        parts: [{ text: "أنت خبير في المطبخ والأواني، مهمتك هي تقديم أفكار إبداعية وملهمة عن وصفات أو طرق تقديم الطعام، بأسلوب جذاب ومختصر." }]
                    },
                };
                
                try {
                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    const result = await response.json();
                    const generatedText = result?.candidates?.[0]?.content?.parts?.[0]?.text;
                    
                    if (generatedText) {
                        ideaOutput.innerText = generatedText;
                    } else {
                        ideaOutput.innerText = 'عذراً، لم أستطع توليد فكرة حالياً. حاول مرة أخرى.';
                    }
                } catch (error) {
                    console.error('API Error:', error);
                    ideaOutput.innerText = 'حدث خطأ. تأكد من اتصالك بالإنترنت.';
                } finally {
                    ideaButton.disabled = false;
                    ideaButton.innerText = 'أعطني فكرة!';
                }
            });
        });
    </script>
